-- Query to find number of professors involved in delivering courses 1 to 10 for each semester and year

select 
s.year, 
s.semester,
count(distinct s.professor_id) as no_of_professor 
from schedule s
where s.course_id between 1 and 10
group by s.year, s.semester
order by s.year asc;
